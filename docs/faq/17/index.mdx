import { FaqPre, FaqPost } from '@site/src/components/FaqCardList'

<FaqPre />

1. 找到「开始」菜单中的「Koishi」项，右键并选择「打开文件位置」
2. 右键出现的快捷方式，选择「属性」
3. 在「目标」一行的前面添加「conhost 」（注意最后有一个空格）
